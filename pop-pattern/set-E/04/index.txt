    let currentTask = { kr: '', en: '' };
    let history = [];
    let playMode = 'manual';
    let autoTimer = null;

    function startTrainer() {
        document.getElementById('start-container').classList.add('hidden');
        document.getElementById('top-controls').classList.add('active');
        document.getElementById('action-area').classList.remove('hidden');
        generate();
    }

    function setPlayMode(mode) {
        playMode = mode;
        document.getElementById('mode-manual').className = `toggle-btn ${mode === 'manual' ? 'active' : ''}`;
        document.getElementById('mode-auto').className = `toggle-btn ${mode === 'auto' ? 'active' : ''}`;
        
        const isAuto = (mode === 'auto');
        document.getElementById('manual-ans-btn').disabled = isAuto;
        document.getElementById('manual-next-btn').disabled = isAuto;
        document.getElementById('manual-ans-btn').style.opacity = isAuto ? "0.5" : "1";
        document.getElementById('manual-next-btn').style.opacity = isAuto ? "0.5" : "1";

        if (isAuto) {
            generate();
        } else {
            clearTimeout(autoTimer);
            updateProgressBar(0, 0);
        }
    }

    function generate() {
        clearTimeout(autoTimer);
        if (currentTask.kr) history.push({...currentTask});
        if (history.length > 50) history.shift();
        const randomIndex = Math.floor(Math.random() * allClaims.length);
        currentTask = allClaims[randomIndex];
        updateDisplay();

        if (playMode === 'auto') {
            runAutoCycle();
        }
    }

    function runAutoCycle() {
        updateProgressBar(100, 5000); 
        autoTimer = setTimeout(() => {
            showAnswer();
            updateProgressBar(100, 3000); 
            autoTimer = setTimeout(() => {
                generate();
            }, 3000);
        }, 5000);
    }

    function updateProgressBar(width, duration) {
        const bar = document.getElementById('auto-progress');
        bar.style.transition = 'none';
        bar.style.width = '0%';
        if(duration > 0) {
            setTimeout(() => {
                bar.style.transition = `width ${duration}ms linear`;
                bar.style.width = '100%';
            }, 10);
        }
    }

    function goBack() {
        if (history.length === 0) return;
        if (playMode === 'auto') setPlayMode('manual');
        currentTask = history.pop();
        updateDisplay();
    }

    function updateDisplay() {
        document.getElementById('korean-task').innerText = currentTask.kr;
        const ans = document.getElementById('english-answer');
        ans.classList.remove('show');
        ans.innerText = currentTask.en;
    }

    function showAnswer() {
        document.getElementById('english-answer').classList.add('show');
    }
